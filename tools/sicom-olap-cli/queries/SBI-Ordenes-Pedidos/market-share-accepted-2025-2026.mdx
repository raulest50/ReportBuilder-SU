        [mayoristas]
        * [target_months]
        * [target_years]
        * [target_products]
        * [target_buyers],
        [Measures].[VOLUMEN ACEPTADO]

    { [Measures].[VOLUMEN ACEPTADO] } ON COLUMNS,
    [rows] ON ROWS
