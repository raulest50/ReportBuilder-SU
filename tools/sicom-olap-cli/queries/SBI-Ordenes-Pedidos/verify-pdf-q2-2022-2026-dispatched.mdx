WITH
SET [target_buyers] AS {
    [COMPRADOR].[SUBTIPO AGENTE COMPRADOR].&[ESTACION DE SERVICIO AUTOMOTRIZ],
    [COMPRADOR].[SUBTIPO AGENTE COMPRADOR].&[ESTACION DE SERVICIO FLUVIAL],
    [COMPRADOR].[SUBTIPO AGENTE COMPRADOR].&[COMERCIALIZADOR INDUSTRIAL]
}
SET [target_products] AS {
    [PRODUCTO].[DESCRIPCION PRODUCTO].&[GASOLINA MOTOR CORRIENTE],
    [PRODUCTO].[DESCRIPCION PRODUCTO].&[BIODIESEL CON MEZCLA],
    [PRODUCTO].[DESCRIPCION PRODUCTO].&[GASOLINA MOTOR EXTRA]
}
SET [target_years] AS {
    [MOVIMIENTOS ORDEN PEDIDO].[AÑO DESPACHO].&[2022],
    [MOVIMIENTOS ORDEN PEDIDO].[AÑO DESPACHO].&[2023],
    [MOVIMIENTOS ORDEN PEDIDO].[AÑO DESPACHO].&[2024],
    [MOVIMIENTOS ORDEN PEDIDO].[AÑO DESPACHO].&[2025],
    [MOVIMIENTOS ORDEN PEDIDO].[AÑO DESPACHO].&[2026]
}
SET [target_months] AS {
    [MOVIMIENTOS ORDEN PEDIDO].[MES DESPACHO].&[4],
    [MOVIMIENTOS ORDEN PEDIDO].[MES DESPACHO].&[5],
    [MOVIMIENTOS ORDEN PEDIDO].[MES DESPACHO].&[6]
}
SET [rows] AS
    NONEMPTY(
        [target_buyers]
        * [target_products]
        * [target_years]
        * [target_months],
        [Measures].[VOLUMEN DESPACHADO]
    )
SELECT
    { [Measures].[VOLUMEN DESPACHADO] } ON COLUMNS,
    [rows] ON ROWS
FROM [Ordenes-Pedidos]
