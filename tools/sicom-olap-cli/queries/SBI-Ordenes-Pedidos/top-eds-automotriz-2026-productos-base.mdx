WITH
SET [target_months] AS {
    [MOVIMIENTOS ORDEN PEDIDO].[MES DESPACHO].&[1],
    [MOVIMIENTOS ORDEN PEDIDO].[MES DESPACHO].&[2],
    [MOVIMIENTOS ORDEN PEDIDO].[MES DESPACHO].&[3],
    [MOVIMIENTOS ORDEN PEDIDO].[MES DESPACHO].&[4],
    [MOVIMIENTOS ORDEN PEDIDO].[MES DESPACHO].&[5],
    [MOVIMIENTOS ORDEN PEDIDO].[MES DESPACHO].&[6]
}
SET [target_products] AS {
    [PRODUCTO].[DESCRIPCION PRODUCTO].&[GASOLINA MOTOR CORRIENTE],
    [PRODUCTO].[DESCRIPCION PRODUCTO].&[GASOLINA MOTOR EXTRA],
    [PRODUCTO].[DESCRIPCION PRODUCTO].&[BIODIESEL CON MEZCLA]
}
SET [top_eds] AS
    TOPCOUNT(
        NONEMPTY(
            [COMPRADOR].[CODIGO SICOM COMPRADOR].[CODIGO SICOM COMPRADOR].MEMBERS
            * [COMPRADOR].[NOMBRE COMERCIAL COMPRADOR].[NOMBRE COMERCIAL COMPRADOR].MEMBERS
            * [COMPRADOR].[MUNICIPIO AGENTE COMPRADOR].[MUNICIPIO AGENTE COMPRADOR].MEMBERS
            * [COMPRADOR].[DEPARTAMENTO AGENTE COMPRADOR].[DEPARTAMENTO AGENTE COMPRADOR].MEMBERS,
            [Measures].[VOLUMEN DESPACHADO]
        ),
        20,
        [Measures].[VOLUMEN DESPACHADO]
    )
SELECT
    {
        [Measures].[VOLUMEN ACEPTADO],
        [Measures].[VOLUMEN DESPACHADO]
    } ON COLUMNS,
    [top_eds] ON ROWS
FROM [Ordenes-Pedidos]
WHERE (
    [MOVIMIENTOS ORDEN PEDIDO].[AÑO DESPACHO].&[2026],
    [COMPRADOR].[SUBTIPO AGENTE COMPRADOR].&[ESTACION DE SERVICIO AUTOMOTRIZ],
    [target_months],
    [target_products]
)
